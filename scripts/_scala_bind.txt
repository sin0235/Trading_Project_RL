%spark
import java.nio.charset.StandardCharsets;
import java.nio.file.{Files, Paths};
import java.util.{LinkedHashMap => JLinkedHashMap, Map => JMap};
import com.fasterxml.jackson.databind.ObjectMapper;

{
  val defaultProjectRoot = Option(z.input("project_root", "/workspace/project"))
    .map(_.toString.trim)
    .filter(s => s.nonEmpty && s != "null")
    .getOrElse("/workspace/project");

  val cacheDir = Paths.get(defaultProjectRoot).toAbsolutePath.normalize.resolve(".zeppelin_cache");
  val dashboardCachePath = cacheDir.resolve("dashboard_payload.json");
  val trainCachePath = cacheDir.resolve("dashboard_train_payload.json");
  val payloadPath = if (Files.exists(dashboardCachePath)) dashboardCachePath else trainCachePath;

  if (!Files.exists(payloadPath)) {
    throw new RuntimeException("Chưa có dashboard cache. Hãy chạy 'Setup Base Payload' trước.");
  }

  val payloadJson = new String(Files.readAllBytes(payloadPath), StandardCharsets.UTF_8);
  val mapper = new ObjectMapper();
  val payloadObj = mapper.readValue(payloadJson, classOf[JLinkedHashMap[_, _]])
    .asInstanceOf[JMap[String, Object]];

  val debugObj = if (payloadObj.get("debug") == null) {
    new JLinkedHashMap[String, Object]()
  } else {
    mapper.convertValue(payloadObj.get("debug"), classOf[JLinkedHashMap[_, _]])
      .asInstanceOf[JMap[String, Object]]
  }
  debugObj.put("bound_json_target_len", Int.box(payloadJson.length));
  val debugJsonStr = mapper.writeValueAsString(debugObj)

  try { z.angularUnbind("rlTradingShowcaseDebugJson") } catch { case _: Throwable => }
  try { z.angularUnbind("rlTradingShowcaseDebug") } catch { case _: Throwable => }
  try { z.angularUnbind("rlTradingShowcase") } catch { case _: Throwable => }
  try { z.angularUnbind("rlTradingShowcaseJson") } catch { case _: Throwable => }
  try { z.angularUnbind("rlTradingShowcaseBuild") } catch { case _: Throwable => }
  try { z.angularUnbind("rlTradingShowcaseJsonLen") } catch { case _: Throwable => }

  val jsonLen = payloadJson.length
  val maxJsonChars = 6000000
  val jsonForBind = if (jsonLen > maxJsonChars) "" else payloadJson

  z.angularBind("rlTradingShowcase", payloadObj);
  z.angularBind("rlTradingShowcaseDebug", debugObj);
  z.angularBind("rlTradingShowcaseDebugJson", debugJsonStr);
  z.angularBind("rlTradingShowcaseJson", jsonForBind);
  z.angularBind("rlTradingShowcaseBuild", s"2026-03-22-fix8-bind:${payloadPath.getFileName.toString}:${jsonLen}");
  z.angularBind("rlTradingShowcaseJsonLen", Int.box(jsonLen));

  val safePath = payloadPath.toString.replace("\\", "\\\\");
  println(s"""{
  "bind_status": "ok",
  "build_tag": "2026-03-22-fix8-bind",
  "payload_path": "${safePath}",
  "json_length": ${jsonLen},
  "json_bind_omitted": ${jsonLen > maxJsonChars},
  "binding_names": ["rlTradingShowcase", "rlTradingShowcaseDebug", "rlTradingShowcaseDebugJson", "rlTradingShowcaseJson", "rlTradingShowcaseBuild", "rlTradingShowcaseJsonLen"]
}""");
  ()
}
